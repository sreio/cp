开奖记录表（增加字段）

-- 双色球开奖记录（更新）
CREATE TABLE ssq_draws (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  issue       TEXT NOT NULL UNIQUE,
  draw_date   DATE NOT NULL,
  week        TEXT,                        -- 星期几 "二"/"四"/"日"
  red_balls   TEXT NOT NULL,               -- 红球 "[01,02,03,04,05,06]"
  blue_ball   TEXT NOT NULL,
  pool_amount DECIMAL(15,2),
  sales_amount DECIMAL(15,2),
  source      TEXT DEFAULT 'official',      -- "official" | "third_party" | "manual"
  created_at  DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at  DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 大乐透开奖记录（更新）
CREATE TABLE dlt_draws (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  issue       TEXT NOT NULL UNIQUE,
  draw_date   DATE NOT NULL,
  week        TEXT,                        -- 星期几 "一"/"三"/"六"
  front_zone  TEXT NOT NULL,
  back_zone   TEXT NOT NULL,
  pool_amount DECIMAL(15,2),
  sales_amount DECIMAL(15,2),
  source      TEXT DEFAULT 'official',
  created_at  DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at  DATETIME DEFAULT CURRENT_TIMESTAMP
);

中奖地区表（新增）

-- 中奖地区详情
CREATE TABLE winning_locations (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  lottery_type TEXT NOT NULL,              -- "ssq" | "dlt"
  issue       TEXT NOT NULL,
  prize_level TEXT NOT NULL,              -- "一等奖"
  location    TEXT NOT NULL,              -- "北京" | "广东" | ...
  win_count   INTEGER NOT NULL,           -- 该地区中奖注数
  win_amount  DECIMAL(15,2),             -- 该地区总奖金
  created_at  DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (lottery_type, issue) REFERENCES draws(type, issue)
);

-- 索引
CREATE INDEX idx_winning_location ON winning_locations(lottery_type, issue);
CREATE INDEX idx_winning_prize ON winning_locations(lottery_type, issue, prize_level);
